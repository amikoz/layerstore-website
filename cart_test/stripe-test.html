🔧 Server Setup

Benötigte PHP-Dateien für Stripe Integration:

create-checkout-session.php:

<?php
require 'vendor/autoload.php';
\Stripe\Stripe::setApiKey('sk_test_123');

header('Content-Type: application/json');

try {
  $checkout_session = \Stripe\Checkout\Session::create([
    'line_items' => [[
      'price_data' => [
        'currency' => 'eur',
        'product_data' => ['name' => '3D Gedruckte Dekoration'],
        'unit_amount' => 2990,
      ],
      'quantity' => 1,
    ]],
    'mode' => 'payment',
    'success_url' => 'http://' . $_SERVER['HTTP_HOST'] . '/layerstore website/cart/stripe-test.html?success=true',
    'cancel_url' => 'http://' . $_SERVER['HTTP_HOST'] . '/layerstore website/cart/stripe-test.html?canceled=true',
  ]);

  echo json_encode(['url' => $checkout_session->url]);
} catch (Exception $e) {
  echo json_encode(['error' => $e->getMessage()]);
}
?>

Für Payment Element:

create-payment-intent.php:

<?php
require 'vendor/autoload.php';
\Stripe\Stripe::setApiKey('sk_test_123');

header('Content-Type: application/json');

$jsonStr = file_get_contents('php://input');
$jsonObj = json_decode($jsonStr);

try {
  $paymentIntent = \Stripe\PaymentIntent::create([
    'amount' => $jsonObj->amount,
    'currency' => $jsonObj->currency,
    'automatic_payment_methods' => ['enabled' => true],
  ]);

  echo json_encode(['clientSecret' => $paymentIntent->client_secret]);
} catch (Exception $e) {
  echo json_encode(['error' => $e->getMessage()]);
}
?>
